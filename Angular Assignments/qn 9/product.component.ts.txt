import { Component, OnInit } from '@angular/core';
import { ProductService } from '../product.service';

@Component({
  selector: 'app-product',
  templateUrl: './product.component.html',
})
export class ProductComponent implements OnInit {
  products: any[] = [];
  saveforlater: any[] = [];

  constructor(private productService: ProductService) {}

  ngOnInit() {
    this.loadProducts();
    this.loadSaveForLater();
  }

  loadProducts() {
    this.productService.getProducts().subscribe((data: any) => this.products = data);
  }

  loadSaveForLater() {
    this.productService.getSaveForLater().subscribe((data: any) => this.saveforlater = data);
  }

  addToSaveForLater(product: any) {
    this.productService.addSaveForLater(product).subscribe(() => this.loadSaveForLater());
  }
}
